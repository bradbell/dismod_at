# get_table
def get_table(name) :
     file_name = name + '.csv'
     file_ptr  = open(file_name, 'r')
     table     = list()
     reader    = csv.DictReader(file_ptr)
     for row in reader :
          table.append(row)
     file_ptr.close()
     return table
#
# near_equal
def near_equal(x, y, eps) :
     if y == 0.0 :
          return float(x) < eps
     return abs( float(x) / y - 1.0 ) <= eps

meas_std          = meas_value / 20.
