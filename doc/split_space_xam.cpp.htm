# include <dismod_at/split_space.hpp>

bool split_space_xam(void)
{
     bool   ok = true;

     std::string str = " one two  three  ";
     CppAD::vector<std::string> vec = dismod_at::split_space(str);

     ok &= vec.size() == 3;
     ok &= vec[0] == "one";
     ok &= vec[1] == "two";
     ok &= vec[2] == "three";

     return ok;
}
